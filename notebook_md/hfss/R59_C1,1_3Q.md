## Ports

```python
"""
Set 8 ports. 2 for readout TL, 3 for charge line and 3 for flux line. 

Port map:
- Top: readout TL.
- Left: Q1 flux and charge lines.
- Bottom: Q2 flux and charge lines.
- Right: Q3 flux and charge lines.
"""
x_port, y_port = 4.500, 4.500

pad_dx = 2.00
pad_dy = 2.00
TL_width, TL_gap = 0.020, 0.012
ctrl_width, ctrl_gap = 0.010, 0.006
launch_lead = 0.100
pad_width, pad_height, taper_height = 0.500, 0.250, 0.500

def make_port(name, pos_x, pos_y, orientation, trace_width, trace_gap):
    offset_dict = {
        "0": (taper_height, 0.0),
        "90": (0.0, pad_width),
        "180": (-taper_height, 0.0),
        "270": (0.0, -pad_width),
    }
    offset = offset_dict.get(orientation, (0, 0))
    return LaunchpadWirebond(
        design,
        name,
        options=dict(  # type: ignore
            pos_x=pos_x + offset[0],
            pos_y=pos_y + offset[1],
            orientation=orientation,
            pad_width=pad_width,
            pad_height=pad_height,
            pad_gap=0.2,
            taper_height=taper_height,
            lead_length=launch_lead,
            trace_width=trace_width,
            trace_gap=trace_gap,
        ),
    )


port_T1 = make_port("port_T1", -pad_dx, y_port-0.2, "180", TL_width, TL_gap)
port_T2 = make_port("port_T2", pad_dx, y_port-0.2, "0", TL_width, TL_gap)

port_L1 = make_port("port_L1", -x_port+0.2, pad_dy, "270", ctrl_width, ctrl_gap)
port_L2 = make_port("port_L2", -x_port, -pad_dy, "0", ctrl_width, ctrl_gap)

port_B1 = make_port("port_B1", -pad_dx, -y_port+0.2, "0", ctrl_width, ctrl_gap)
port_B2 = make_port("port_B2", pad_dx, -y_port+0.2, "180", ctrl_width, ctrl_gap)

port_R1 = make_port("port_R1", x_port-0.2, pad_dy, "270", ctrl_width, ctrl_gap)
port_R2 = make_port("port_R2", x_port, -pad_dy, "180", ctrl_width, ctrl_gap)

gui.rebuild()
gui.autoscale()
gui.screenshot()
```
